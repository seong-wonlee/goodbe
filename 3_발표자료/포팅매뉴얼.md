```yaml

	url: jdbc:h2:tcp://localhost/~/goodbe

	password:
	driver-class-name: org.h2.Driver
```

- remote - application-prod.yml

```yaml
datasource:
	driver-class-name: org.mariadb.jdbc.Driver
	url: jdbc:mariadb://[url]:3306/[DB이름]
	username: [DB유저이름]
	password: [비밀번호]
```
